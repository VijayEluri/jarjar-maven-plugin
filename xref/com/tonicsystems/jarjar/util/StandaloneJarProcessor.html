27  public class StandaloneJarProcessor
28  {
29      public static void run(File from, File to, JarProcessor proc) throws IOException {
30          run(from, to, proc, false);
31      }
32      public static void run(File from, File to, JarProcessor proc, boolean ignoreDuplicates) throws IOException {
33          byte[] buf = new byte[0x2000];
34          File tmp = null;
35          if (from.equals(to)) {
36              tmp = File.createTempFile("jarjar", null);
37              IoUtil.copy(from, tmp, buf);
38              from = tmp;
39          }
40          JarFile in = new JarFile(from);
41          JarOutputStream out = new JarOutputStream(new FileOutputStream(to));
42          Set<String> entries = new HashSet<String>();
43          try {
44              EntryStruct struct = new EntryStruct();
45              Enumeration<JarEntry> e = in.entries();
46              while (e.hasMoreElements()) {
47                  JarEntry entry = e.nextElement();
48                  struct.name = entry.getName();
49                  struct.time = entry.getTime();
50                  ByteArrayOutputStream baos = new ByteArrayOutputStream();
51                  IoUtil.pipe(in.getInputStream(entry), baos, buf);
52                  struct.data = baos.toByteArray();
53                  if (proc.process(struct)) {
54                      if (entries.add(struct.name)) {
55                          entry = new JarEntry(struct.name);
56                          entry.setTime(struct.time);
57                          entry.setCompressedSize(-1);
58                          out.putNextEntry(entry);
59                          out.write(struct.data);
60                      } else if (struct.name.endsWith("/")) {
61                          // TODO(chrisn): log
62                      } else if (!ignoreDuplicates) {
63                          throw new IllegalArgumentException("Duplicate jar entries: " + struct.name);
64                      }
65                  }
66              }
67          } finally {
68              out.close();
69              if (tmp != null)
70                  tmp.delete();
71          }
72      }
