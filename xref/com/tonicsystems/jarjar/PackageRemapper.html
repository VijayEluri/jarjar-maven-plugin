17  package com.tonicsystems.jarjar;
18  
19  import com.tonicsystems.jarjar.util.*;
20  import org.objectweb.asm.*;
21  import org.objectweb.asm.signature.*;
22  import org.objectweb.asm.commons.*;
23  import java.util.*;
24  import java.util.regex.Pattern;
25  
26  class PackageRemapper extends Remapper
27  {
28      private static final String RESOURCE_SUFFIX = "RESOURCE";
29      
30      private static final Pattern ARRAY_FOR_NAME_PATTERN
31          = Pattern.compile("\\[L[\\p{javaJavaIdentifierPart}\\.]+?;");
32  
33      private final List<Wildcard> wildcards;
34      private final Map<String, String> typeCache = new HashMap<String, String>();
35      private final Map<String, String> pathCache = new HashMap<String, String>();
36      private final Map<Object, String> valueCache = new HashMap<Object, String>();
37      private final boolean verbose;
38  
39      public PackageRemapper(List<Rule> ruleList, boolean verbose) {
40          this.verbose = verbose;
41          wildcards = PatternElement.createWildcards(ruleList);
42      }
43  
44      // also used by KeepProcessor
45      static boolean isArrayForName(String value) {
46        return ARRAY_FOR_NAME_PATTERN.matcher(value).matches();
47      }
48  
49      public String map(String key) {
50          String s = typeCache.get(key);
51          if (s == null) {
52              s = replaceHelper(key);
53              if (key.equals(s))
54                  s = null;
55              typeCache.put(key, s);
56          }
57          return s;
58      }
59  
60      public String mapPath(String path) {
61          String s = pathCache.get(path);
62          if (s == null) {
63              s = path;
64              int slash = s.lastIndexOf('/');
65              String end;
66              if (slash < 0) {
67                  end = s;
68                  s = RESOURCE_SUFFIX;
69              } else {
70                  end = s.substring(slash + 1);
71                  s = s.substring(0, slash + 1) + RESOURCE_SUFFIX;
72              }
73              boolean absolute = s.startsWith("/");
74              if (absolute)
75                  s = s.substring(1);
76              try {
77                s = mapType(s);
78              } catch (Exception e) {
79                // s was not a valid type
80              }
81              if (absolute)
82                  s = "/" + s;
83              if (s.indexOf(RESOURCE_SUFFIX) < 0)
84                return path;
85              s = s.substring(0, s.length() - RESOURCE_SUFFIX.length()) + end;
86              pathCache.put(path, s);
87          }
88          return s;
89      }
90  
91      public Object mapValue(Object value) {
92          if (value instanceof String) {
93              String s = valueCache.get(value);
94              if (s == null) {
95                  s = (String)value;
96                  if (isArrayForName(s)) {
97                      String desc1 = s.replace('.', '/');
98                      String desc2 = mapDesc(desc1);
99                      if (!desc2.equals(desc1))
100                         return desc2.replace('/', '.');
101                 } else {
102                     s = mapPath(s);
103                     if (s.equals(value)) {
104                         boolean hasDot = s.indexOf('.') >= 0;
105                         boolean hasSlash = s.indexOf('/') >= 0;
106                         if (!(hasDot && hasSlash)) {
107                             if (hasDot) {
108                                 s = replaceHelper(s.replace('.', '/')).replace('/', '.');
109                             } else {
110                                 s = replaceHelper(s);
111                             }
112                         }
113                     }
114                 }
115                 valueCache.put(value, s);
116             }
117             // TODO: add back class name to verbose message
118             if (verbose && !s.equals(value))
119                 System.err.println("Changed \"" + value + "\" -> \"" + s + "\"");
120             return s;
121         } else {
122             return super.mapValue(value);
123         }
124     }
125 
126     private String replaceHelper(String value) {
127         for (Wildcard wildcard : wildcards) {
128             String test = wildcard.replace(value);
129             if (test != null)
130                 return test;
131         }
132         return value;
133     }
